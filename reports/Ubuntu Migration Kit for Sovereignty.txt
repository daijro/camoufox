INTERFACE = "eth0" # Change to wlan0 if on WiFi

def load_ebpf_mask():
    if os.geteuid() != 0:
        print("[FAIL] eBPF requires ROOT. Run with sudo.")

4. Hardware Sovereignty (vTPM Manager)
Ubuntu allows us to spawn a software TPM (swtpm) easily. This script manages a unique TPM chip for each browser profile.
import subprocess
import os
import time


class TPMManager:
    def __init__(self, profile_id):
        self.profile_id = profile_id
        self.tpm_state_dir = f"profiles/{profile_id}/tpm_state"
        self.sock_path = f"/tmp/swtpm-{profile_id}.sock"
        self.process = None


    def start_tpm(self):
        """
        Spawns a Software TPM process for this specific profile.
        """
        if not os.path.exists(self.tpm_state_dir):
            os.makedirs(self.tpm_state_dir)


        print(f"[{self.profile_id}] HARDWARE: Initializing vTPM Enclave...")
        
        cmd = [
            "swtpm", "socket",
            "--tpmstate", f"dir={self.tpm_state_dir}",
            "--ctrl", f"type=unixio,path={self.sock_path}",
            "--tpm2",
            "--log", "level=0"
        ]
        
        try:
            # Spawn in background
            self.process = subprocess.Popen(cmd)
            time.sleep(1) # Wait for socket
            
            if os.path.exists(self.sock_path):
                print(f"[{self.profile_id}] HARDWARE: vTPM Active. Socket: {self.sock_path}")
                return self.sock_path
            else:
                print(f"[{self.profile_id}] HARDWARE: Failed to create socket.")
                return None
                
        except FileNotFoundError:
            print("[ERROR] 'swtpm' not installed. Run 'sudo apt install swtpm'.")
            return None


    def stop_tpm(self):
        if self.process:
            self.process.terminate()
            print(f"[{self.profile_id}] HARDWARE: vTPM Halted.")
